  • Sass
    $n: 12;
$g: .25em;
$w: .5em;
$b: ();
$s: ();
$a: 360/$n;

@for $i from 1 through $n {
  $c: hsl($i*$a, 100%, 50%);
  $b: $b, linear-gradient($c, $c) 
    ($i - 1)*($w + $g) 100%;
  $s: $s, $w (1 + sin($a*1deg))*50%;
}

.wave {
  background: $b;
  background-repeat: no-repeat;
  background-size: $s;
}

the code for the animated wave 

$n: 12;
$p: $n;
$t: 2;
.wave { animation: wave-ani $t*1s linear infinite; }
@keyframes wave-ani {
  @for $j from 1 through $p {
    #{$j*100%/$p} {
      $s: ();
      @for $i from 1 through $n {
        $s: $s, 
          .5em (1 + sin(($j*$t/$p + $i/$n)*360deg))*50%;
      }
      background-size: $s;
    }
  }
}

putting it all together 

$n: 40;
$a: 360deg/$n;
$R: 8em;
$r1: $R/(2*cos($a/2));
$bg: ();
$cover: ();
@for $i from 0 to $n {
  $bg: $bg, radial-gradient(circle 
    at $R + $r1*cos($i*$a) $R + $r1*sin($i*$a), 
    hsl(0, 0%, (1 - pow(-1, $i))*50%) $r1, transparent $r1);
  $cover: $cover, radial-gradient(circle 
    at $R + $r1*cos($i*$a - 180deg) 
       $R + $r1*sin($i*$a - 180deg), 
    hsl(0, 0%, (1 - pow(-1, $i))*50%) $r1, transparent $r1);
}
